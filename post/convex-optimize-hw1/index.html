代码

function [x,r] = circle_fit(A)
    b=diag(A'*A);
    A_tilde=[2*A',-ones(size(b))];
    y=(A_tilde'*A_tilde)\(A_tilde'*b);
    x=y(1:2,:);
    r=sqrt(x'*x-y(3));
end
